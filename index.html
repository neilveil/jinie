Examples
Simple image editor
<button
  onClick={() =>
    Jinie.init({
      accept: 'image/jpeg',
      onResult: ({ code, img }) => {
        console.log(code, img)
        if (img) setImgURL(window.URL.createObjectURL(img))
      }
    })
  }
>
  Upload Image
</button>
Upload with compression
<button
  onClick={() =>
    Jinie.init({
      accept: 'image/jpeg',
      onResult: ({ img }) => {
        if (img) setImgURL(window.URL.createObjectURL(img))
      },
      maxSize: 2 * 1024
    })
  }
>
  Upload Image
</button>
Upload square png image with transparent background
<button
  onClick={() =>
